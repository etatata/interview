### 解题思路
暴力破解
1，先检查9个小方格是否重复。
2，然后循环检查每个月元素的横和竖列是否重复的。
检查的时候用了一个小技巧。
1，设置一个检查数组，数组的角标位元素的值，数组的值位是否存在，如果存在了1.如果下次循环又遇到1了。就是表明重复了。直接返回false；

### 代码

```java
class Solution {
    public boolean isValidSudoku(char[][] board) {
        //先检查所有小方格。
        int[][] nums = new int[][]{
                {0, 0}, {0, 3}, {0, 6},
                {3, 0}, {3, 3}, {3, 6},
                {6, 0}, {6, 3}, {6, 6}};

        int[] checkNum = new int[10];
        int m, n , k;
        for (k = 0; k < nums.length; k++) {
             Arrays.fill(checkNum,0);
            for (m = nums[k][0]; m <= nums[k][0] + 2 && m >= 0 && m < board.length; m++) {
              
                for (n = nums[k][1]; n <= nums[k][1] + 2 && n >= 0 && n < board[m].length; n++) {
                    if(board[m][n] == '.')continue;
                    if(checkNum[board[m][n] - '0'] == 0){
                        checkNum[board[m][n] - '0'] = 1;
                    }else if(checkNum[board[m][n] - '0'] == 1){
                        return false;
                    }

                }
            }
        }


        for (int i = 0; i < board.length; i++) {
            for (int j = 0; j < board[i].length; j++) {
                if (board[i][j] == '.') {
                    continue;
                }
                if (!check(i, j, board)) {
                    return false;
                }
            }
        }

        return true;
    }


    private boolean check(int i, int j, char[][] board) {


        //1,检查横
        char x = board[i][j];
        for (int m = 0; m < board.length; m++) {
            if (m == j) continue;
            if (board[i][m] == x) {
                return false;
            }
        }

        //2，检查竖列

        for (int m = 0; m < board.length; m++) {
            if (m == i) continue;
            if (board[m][j] == x) {
                return false;
            }
        }

        //3，检查9宫格
        //0,0   0,3  0,6
        //3,0,  3,3  3,6
        //6,0,  6,3, 6,6


        return true;

    }
    
}
```
